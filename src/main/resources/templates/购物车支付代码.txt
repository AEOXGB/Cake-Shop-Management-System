<script type="text/javascript">
var currentPage = '${currentPage}';//现在所在的页码
var totalCount = '${totalCount}';//产品总数量
var pageCount = '${pageCount}';//总页数

//首页
function pageHome(){
	if (currentPage=='0'){
        new $.zui.Messager("已经是首页!", {
		   placement:'center',
		   icon: 'bell',
		   type: 'warning'
		}).show();
	}
	else{
		location.href = "${pageContext.request.contextPath}/cart/cartList.action?currentPage=0";
	}
}
//上一页
function pagePre(){
	if (currentPage=='0'){
		new $.zui.Messager("已经是第一页!", {
			   placement:'center',
			   icon: 'bell',
			   type: 'warning'
			}).show();
	}
	else{
		location.href = '${pageContext.request.contextPath}/cart/cartList.action?currentPage='+(parseInt(currentPage)-1);
	}
}
//下一页
function pageNext(){
	if (parseInt(currentPage)+1==pageCount){
		new $.zui.Messager("已经是最后一页!", {
			   placement:'center',
			   icon: 'bell',
			   type: 'warning'
			}).show();
	}
	else{
		location.href = '${pageContext.request.contextPath}/cart/cartList.action?currentPage='+(parseInt(currentPage)+1);
	}
}
//尾页
function pageEnd(){
	if (parseInt(currentPage)+1==pageCount){
		new $.zui.Messager("已经是最后一页!", {
			   placement:'center',
			   icon: 'bell',
			   type: 'warning'
			}).show();
	}
	else{
		location.href = '${pageContext.request.contextPath}/cart/cartList.action?currentPage='+(parseInt(pageCount)-1);
	}
}

function add(goodid){
	console.log("goodid"+goodid);
	location.href = '${pageContext.request.contextPath}/cart/add.action?goodid='+goodid;
}

function sub(goodid){
	location.href = '${pageContext.request.contextPath}/cart/sub.action?goodid='+goodid;
}

function deleteGood(goodid){
	//删除之前先询问
	if(confirm("确定要删除此商品？")){
		location.href = '${pageContext.request.contextPath}/cart/delete.action?goodid='+goodid;
	}
	
}
